# Honored to Serve

	lifestyle = stewardship_lifestyle
	tree = duty
	position = { 1 0 }
	icon = node_stewardship
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_stewardship_trigger = yes
			}
			add = 1989
		}
